#!/bin/sh
#PBS -N 0195_intg_eval
#PBS -P gcg51557
#PBS -q R9920261000
#PBS -v RTYPE=rt_HG
#PBS -l select=1
#PBS -l walltime=24:00:00
#PBS -o /nonexistent/eval-out/logs/qsub.out
#PBS -e /nonexistent/eval-out/logs/qsub.err


set -ux

# FIXME: Model name or **absolute path** to the model
MODEL_NAME_OR_PATH="/models/dummy"
# FIXME: **Absolute path** to the output directory
OUTPUT_DIR="/nonexistent/eval-out"

LOG_DIR="${OUTPUT_DIR}/logs"
mkdir -p $LOG_DIR

export HF_HOME="/groups/gcg51557/experiments/dummy/.cache/huggingface"
export HF_TOKEN="dummy-hf-token"
export NUM_GPU=$(nvidia-smi --query-gpu=name --format=csv,noheader | wc -l)
export CUDA_VISIBLE_DEVICES=$(seq -s, 0 $((NUM_GPU-1)))
export NUMEXPR_MAX_THREADS=192

pushd /groups/gcg51557/experiments/0230_intg_eval_2509/environment

# Run all evaluations against a single shared vLLM server (the model is
# loaded exactly once per job). Requires the vllm-serve scripts to be
# installed under /groups/gcg51557/experiments/0230_intg_eval_2509/environment/vllm-serve (see
# installers/intg-eval-2508/vllm-serve/README.md).
bash /groups/gcg51557/experiments/0230_intg_eval_2509/environment/vllm-serve/run_eval_serve.sh \
    $MODEL_NAME_OR_PATH \
    $OUTPUT_DIR \
    --experiment-dir /groups/gcg51557/experiments/0230_intg_eval_2509 \
    --tensor-parallel-size 1 \
    --gpu-memory-utilization 0.9 \
    --swallow --swallow-env swallow_v202411 \
    --llm-jp-eval-versions v2.1.5 \
    --max-num-samples 100 \
    --basemodel
exit_code: 0
